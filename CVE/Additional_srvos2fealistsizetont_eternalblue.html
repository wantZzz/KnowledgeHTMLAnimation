ULONG SrvOs2FeaListSizeToNt(PFEA_LIST pFeaListA) {
// 第一階段：設定基準 A 指標
PFEA pFeaB = pFeaListA->List;
ULONG TotalNtSize = 0;
// 第二階段：迴圈掃描 OS/2 FEA 並累算 NT 大小
while ((PCHAR)pFeaB < (PCHAR)pFeaListA + pFeaListA->cbList) {
TotalNtSize += Padding(sizeof(FILE_FULL_EA_INFORMATION) + pFeaB->cbName + pFeaB->cbValue);
pFeaB = (PFEA)((PCHAR)pFeaB + sizeof(FEA) + pFeaB->cbName + 1 + pFeaB->cbValue);
}
// 第三階段：利用位址相減，修正並以小端序回寫至 cbList 欄位的前 4 個位元組
*(PWORD)pFeaListA = (WORD)((PCHAR)pFeaB - (PCHAR)pFeaListA);
return TotalNtSize;
}
