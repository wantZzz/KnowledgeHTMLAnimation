// 翻譯器主啟動類別
class Postfix {
    public static void main(String[] args) {
        Parser parser = new Parser(); // 建立分析器，預讀第一個字元
        parser.expr(); // 呼叫非終止符號 expr 進入翻譯流程
        System.out.println(); // 輸出換行符號，結束翻譯程序
    }
}
// 非終止符號 E → T R 實作程序
void expr() {
    term(); // 1. 解析最左側的項 T (digit)
    rest(); // 2. 解析運算式餘項 R (加減法尾隨)
}
// 非終止符號 R → + T {print('+')} R | - T {print('-')} R | ε
void rest() {
    if (lookahead == '+') {
        match('+'); // 匹配並消耗 '+'
        term(); // 剖析加號右側運算元
        System.out.print('+'); // 語意動作：印出 '+' 實現後序化
        rest(); // 尾部遞迴，解析剩餘部分
    } else if (lookahead == '-') {
        match('-'); // 匹配並消耗 '-'
        term(); // 剖析減號右側運算元
        System.out.print('-'); // 語意動作：印出 '-' 實現後序化
        rest(); // 尾部遞迴
    } else {
        // ε-產生式：預設不執行任何動作，直接返回退出
    }
}
// 終端符號匹配與讀取
void term() {
    if (Character.isDigit((char)lookahead)) {
        System.out.print((char)lookahead); // 語意動作：讀到數字直接輸出
        match(lookahead); // 執行匹配與指標推進
    } else {
        throw new Error("語法錯誤");
    }
}
// 檢查當前 lookahead 是否相符，並推進指標
void match(int t) {
    if (lookahead == t) {
        lookahead = System.in.read(); // 讀取串流中下一個字元
    } else {
        throw new Error("語法錯誤");
    }
}
